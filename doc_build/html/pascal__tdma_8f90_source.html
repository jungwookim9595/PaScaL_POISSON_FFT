189 subroutine pascal_tdma_plan_many_create(plan, n_sys, myrank, nprocs, mpi_world)
190
191 implicit none
192
193 type(ptdma_plan_many), intent(inout) :: plan
194 integer, intent(in) :: n_sys
195 integer, intent(in) :: myrank, nprocs, mpi_world
196
197 integer :: i, ierr
198 integer :: ista, iend ! First and last indices of assigned range in many tridiagonal systems of equations
199 integer :: bigsize(2), subsize(2), start(2) ! Temporary variables of derived data type (DDT)
200 integer :: ns_rd, nr_rd ! Dimensions of many reduced tridiagonal systems
201 integer :: ns_rt, nr_rt ! Dimensions of many reduced tridiagonal systems after transpose
202 integer, allocatable, dimension(:):: ns_rt_array ! Array specifying the number of tridiagonal systems for each process after transpose
203
204 plan%nprocs = nprocs
205
206 ! Specify dimensions for reduced systems.
207 ns_rd = n_sys
208 nr_rd = 2
209
210 ! Specify dimensions for reduced systems after transpose.
211 ! ns_rt : divide the number of tridiagonal systems of equations per each process
212 ! ns_rt_array : save the ns_rt in ns_rt_array for defining the DDT
213 ! nr_rt : dimensions of the reduced tridiagonal systems in the solving direction, nr_rd*nprocs
214 call para_range(1, ns_rd, nprocs, myrank, ista, iend)
215 ns_rt = iend - ista + 1
216 allocate(ns_rt_array(0:nprocs-1))
217 call mpi_allgather(ns_rt, 1, mpi_integer, ns_rt_array,1, mpi_integer, mpi_world, ierr)
218 nr_rt = nr_rd*nprocs
219
220 ! Assign plan variables and allocate coefficient arrays.
221 plan%n_sys_rt = ns_rt
222 plan%n_row_rt = nr_rt
223 plan%ptdma_world = mpi_world
224
225 allocate( plan%A_rd(1:ns_rd, 1:nr_rd) )
226 allocate( plan%B_rd(1:ns_rd, 1:nr_rd) )
227 allocate( plan%C_rd(1:ns_rd, 1:nr_rd) )
228 allocate( plan%D_rd(1:ns_rd, 1:nr_rd) )
229 allocate( plan%A_rt(1:ns_rt, 1:nr_rt) )
230 allocate( plan%B_rt(1:ns_rt, 1:nr_rt) )
231 allocate( plan%C_rt(1:ns_rt, 1:nr_rt) )
232 allocate( plan%D_rt(1:ns_rt, 1:nr_rt) )
233
234 ! Building the DDTs.
235 allocate(plan%ddtype_Fs(0:nprocs-1), plan%ddtype_Bs(0:nprocs-1))
236
237 do i=0,nprocs-1
238 ! DDT for sending coefficients of the reduced tridiagonal systems using MPI_Ialltoallw communication.
239 bigsize(1)=ns_rd
240 bigsize(2)=nr_rd
241 subsize(1)=ns_rt_array(i)
242 subsize(2)=nr_rd
243 start(1)=sum(ns_rt_array(0:i)) - ns_rt_array(i)
244 start(2)=0
245 call mpi_type_create_subarray( 2, bigsize, subsize, start, &
246 mpi_order_fortran, mpi_double_precision, &
247 plan%ddtype_Fs(i), ierr )
248 call mpi_type_commit(plan%ddtype_Fs(i), ierr)
249 ! DDT for receiving coefficients for the transposed systems of reduction using MPI_Ialltoallw communication.
250 bigsize(1)=ns_rt
251 bigsize(2)=nr_rt
252 subsize(1)=ns_rt
253 subsize(2)=nr_rd
254 start(1)=0
255 start(2)=nr_rd*i
256 call mpi_type_create_subarray( 2, bigsize, subsize, start, &
257 mpi_order_fortran, mpi_double_precision, &
258 plan%ddtype_Bs(i), ierr )
259 call mpi_type_commit(plan%ddtype_Bs(i), ierr)
260 enddo
261
262 ! Buffer counts and displacements for MPI_Ialltoallw.
263 ! All buffer counts are 1 and displacements are 0 due to the defined DDT.
264 allocate(plan%count_send(0:nprocs-1), plan%displ_send(0:nprocs-1))
265 allocate(plan%count_recv(0:nprocs-1), plan%displ_recv(0:nprocs-1))
266 plan%count_send=1; plan%displ_send=0
267 plan%count_recv=1; plan%displ_recv=0
268
269 ! Deallocate local array.
270 if(allocated(ns_rt_array)) deallocate(ns_rt_array)
271
272 end subroutine pascal_tdma_plan_many_create

307 subroutine pascal_tdma_single_solve(plan, A, B, C, D, n_row)
308
309 implicit none
310
311 type(ptdma_plan_single), intent(inout) :: plan
312 double precision, intent(inout) :: a(1:n_row), b(1:n_row), c(1:n_row), d(1:n_row)
313 integer, intent(in) :: n_row
314
315 ! Temporary variables for computation and parameters for MPI functions
316 double precision :: r
317 integer :: i
318 integer :: request(4), ierr
319
320 if (plan%nprocs.eq.1) then
321
322 ! Solve the tridiagonal systems directly when nprocs = 1.
323 call tdma_single(a, b, c, d, n_row)
324 else
325 ! Reduction step : elimination of lower diagonal elements
326 a(1) = a(1)/b(1)
327 d(1) = d(1)/b(1)
328 c(1) = c(1)/b(1)
329
330 a(2) = a(2)/b(2)
331 d(2) = d(2)/b(2)
332 c(2) = c(2)/b(2)
333
334 do i=3,n_row
335 r = 1.d0/(b(i)-a(i)*c(i-1))
336 d(i) = r*(d(i)-a(i)*d(i-1))
337 c(i) = r*c(i)
338 a(i) = -r*a(i)*a(i-1)
339 enddo
340
341 ! Reduction step : elimination of upper diagonal elements
342 do i=n_row-2,2,-1
343 d(i) = d(i)-c(i)*d(i+1)
344 a(i) = a(i)-c(i)*a(i+1)
345 c(i) =-c(i)*c(i+1)
346 enddo
347
348 r = 1.d0/(1.d0-a(2)*c(1))
349 d(1) = r*(d(1)-c(1)*d(2))
350 a(1) = r*a(1)
351 c(1) = -r*c(1)*c(2)
352
353 ! Construct a reduced tridiagonal system of equations per each rank. Each process has two reduced rows.
354 plan%A_rd(1) = a(1); plan%A_rd(2) = a(n_row)
355 plan%B_rd(1) = 1.d0; plan%B_rd(2) = 1.d0
356 plan%C_rd(1) = c(1); plan%C_rd(2) = c(n_row)
357 plan%D_rd(1) = d(1); plan%D_rd(2) = d(n_row)
358
359 ! Gather the coefficients of the reduced tridiagonal system to a defined rank, plan%gather_rank.
360 call mpi_igather(plan%A_rd, 2, mpi_double_precision, &
361 plan%A_rt, 2, mpi_double_precision, &
362 plan%gather_rank, plan%ptdma_world, request(1), ierr)
363 call mpi_igather(plan%B_rd, 2, mpi_double_precision, &
364 plan%B_rt, 2, mpi_double_precision, &
365 plan%gather_rank, plan%ptdma_world, request(2), ierr)
366 call mpi_igather(plan%C_rd, 2, mpi_double_precision, &
367 plan%C_rt, 2, mpi_double_precision, &
368 plan%gather_rank, plan%ptdma_world, request(3), ierr)
369 call mpi_igather(plan%D_rd, 2, mpi_double_precision, &
370 plan%D_rt, 2, mpi_double_precision, &
371 plan%gather_rank, plan%ptdma_world, request(4), ierr)
372 call mpi_waitall(4, request, mpi_statuses_ignore, ierr)
373
374 ! Solve the reduced tridiagonal system on plan%gather_rank.
375 if(plan%myrank == plan%gather_rank) then
376 call tdma_single(plan%A_rt,plan%B_rt,plan%C_rt,plan%D_rt, plan%n_row_rt)
377 endif
378
379 ! Scatter the solutions to each rank.
380 call mpi_iscatter(plan%D_rt, 2, mpi_double_precision, &
381 plan%D_rd, 2, mpi_double_precision, &
382 plan%gather_rank, plan%ptdma_world, request(1), ierr)
383
384 call mpi_waitall(1, request, mpi_statuses_ignore, ierr)
385
386 ! Update solutions of the modified tridiagonal system with the solutions of the reduced tridiagonal system.
387 d(1 ) = plan%D_rd(1)
388 d(n_row) = plan%D_rd(2)
389 do i=2,n_row-1
390 d(i) = d(i)-a(i)*d(1)-c(i)*d(n_row)
391 enddo
392 endif
393
394 end subroutine pascal_tdma_single_solve

506 subroutine pascal_tdma_many_solve(plan, A, B, C, D, n_sys, n_row)
507
508 implicit none
509
510 type(ptdma_plan_many), intent(inout) :: plan
511 double precision, intent(inout) :: a(1:n_sys,1:n_row), b(1:n_sys,1:n_row), c(1:n_sys,1:n_row), d(1:n_sys,1:n_row)
512 integer, intent(in) :: n_sys, n_row
513
514 ! Temporary variables for computation and parameters for MPI functions.
515 integer :: i, j
516 integer :: request(4),ierr
517 double precision :: r
518
519 if (plan%nprocs.eq.1) then
520
521 ! Solve the tridiagonal systems directly when nprocs = 1.
522 call tdma_many(a, b, c, d, n_sys, n_row)
523 else
524
525 ! The modified Thomas algorithm : elimination of lower diagonal elements.
526 ! First index indicates a number of independent many tridiagonal systems to use vectorization.
527 ! Second index indicates a row number in a partitioned tridiagonal system .
528 do i=1, n_sys
529 a(i,1) = a(i,1)/b(i,1)
530 d(i,1) = d(i,1)/b(i,1)
531 c(i,1) = c(i,1)/b(i,1)
532
533 a(i,2) = a(i,2)/b(i,2)
534 d(i,2) = d(i,2)/b(i,2)
535 c(i,2) = c(i,2)/b(i,2)
536 enddo
537
538 do j=3, n_row
539 do i=1, n_sys
540 r = 1.d0/(b(i,j)-a(i,j)*c(i,j-1))
541 d(i,j) = r*(d(i,j)-a(i,j)*d(i,j-1))
542 c(i,j) = r*c(i,j)
543 a(i,j) = -r*a(i,j)*a(i,j-1)
544 enddo
545 enddo
546
547 ! The modified Thomas algorithm : elimination of upper diagonal elements.
548 do j=n_row-2, 2, -1
549 do i=1, n_sys
550 d(i,j) = d(i,j)-c(i,j)*d(i,j+1)
551 a(i,j) = a(i,j)-c(i,j)*a(i,j+1)
552 c(i,j) =-c(i,j)*c(i,j+1)
553 enddo
554 enddo
555
556 do i=1, n_sys
557 r = 1.d0/(1.d0-a(i,2)*c(i,1))
558 d(i,1) = r*(d(i,1)-c(i,1)*d(i,2))
559 a(i,1) = r*a(i,1)
560 c(i,1) = -r*c(i,1)*c(i,2)
561
562 ! Construct many reduced tridiagonal systems per each rank. Each process has two rows of reduced systems.
563 plan%A_rd(i,1) = a(i,1); plan%A_rd(i,2) = a(i,n_row)
564 plan%B_rd(i,1) = 1.d0 ; plan%B_rd(i,2) = 1.d0
565 plan%C_rd(i,1) = c(i,1); plan%C_rd(i,2) = c(i,n_row)
566 plan%D_rd(i,1) = d(i,1); plan%D_rd(i,2) = d(i,n_row)
567 enddo
568
569
570 ! Transpose the reduced systems of equations for TDMA using MPI_Ialltoallw and DDTs.
571 call mpi_ialltoallw(plan%A_rd, plan%count_send, plan%displ_send, plan%ddtype_Fs, &
572 plan%A_rt, plan%count_recv, plan%displ_recv, plan%ddtype_Bs, &
573 plan%ptdma_world, request(1), ierr)
574 call mpi_ialltoallw(plan%B_rd, plan%count_send, plan%displ_send, plan%ddtype_Fs, &
575 plan%B_rt, plan%count_recv, plan%displ_recv, plan%ddtype_Bs, &
576 plan%ptdma_world, request(2), ierr)
577 call mpi_ialltoallw(plan%C_rd, plan%count_send, plan%displ_send, plan%ddtype_Fs, &
578 plan%C_rt, plan%count_recv, plan%displ_recv, plan%ddtype_Bs, &
579 plan%ptdma_world, request(3), ierr)
580 call mpi_ialltoallw(plan%D_rd, plan%count_send, plan%displ_send, plan%ddtype_Fs, &
581 plan%D_rt, plan%count_recv, plan%displ_recv, plan%ddtype_Bs, &
582 plan%ptdma_world, request(4), ierr)
583
584 call mpi_waitall(4, request, mpi_statuses_ignore, ierr)
585
586 ! Solve the reduced tridiagonal systems of equations using Thomas algorithm.
587 call tdma_many(plan%A_rt,plan%B_rt,plan%C_rt,plan%D_rt, plan%n_sys_rt, plan%n_row_rt)
588
589 ! Transpose the obtained solutions to original reduced forms using MPI_Ialltoallw and DDTs.
590 call mpi_ialltoallw(plan%D_rt, plan%count_recv, plan%displ_recv, plan%ddtype_Bs, &
591 plan%D_rd, plan%count_send, plan%displ_send, plan%ddtype_Fs, &
592 plan%ptdma_world, request(1), ierr)
593 call mpi_waitall(1, request, mpi_statuses_ignore, ierr)
594
595 ! Update solutions of the modified tridiagonal system with the solutions of the reduced tridiagonal system.
596 do i=1,n_sys
597 d(i,1 ) = plan%D_rd(i,1)
598 d(i,n_row) = plan%D_rd(i,2)
599 enddo
600
601 do j=2,n_row-1
602 do i=1,n_sys
603 d(i,j) = d(i,j)-a(i,j)*d(i,1)-c(i,j)*d(i,n_row)
604 enddo
605 enddo
606 endif
607
608 end subroutine pascal_tdma_many_solve

660 ! The modified Thomas algorithm : elimination of upper diagonal elements.
661 do j=n_row-2,2,-1
662 do i=1,n_sys
663 d(i,j) = d(i,j)-c(i,j)*d(i,j+1)
664 a(i,j) = a(i,j)-c(i,j)*a(i,j+1)
665 c(i,j) =-c(i,j)*c(i,j+1)
666 enddo
667 enddo
668
669 do i=1,n_sys
670 r = 1.d0/(1.d0-a(i,2)*c(i,1))
671 d(i,1) = r*(d(i,1)-c(i,1)*d(i,2))
672 a(i,1) = r*a(i,1)
673 c(i,1) = -r*c(i,1)*c(i,2)
674
675 ! Construct the reduced tridiagonal equations per each rank. Each process has two rows of reduced systems.
676 plan%A_rd(i,1) = a(i,1); plan%A_rd(i,2) = a(i,n_row)
677 plan%B_rd(i,1) = 1.d0 ; plan%B_rd(i,2) = 1.d0
678 plan%C_rd(i,1) = c(i,1); plan%C_rd(i,2) = c(i,n_row)
679 plan%D_rd(i,1) = d(i,1); plan%D_rd(i,2) = d(i,n_row)
680 enddo
681
682 ! Transpose the reduced systems of equations for TDMA using MPI_Ialltoallw and DDTs.
683 call mpi_ialltoallw(plan%A_rd, plan%count_send, plan%displ_send, plan%ddtype_Fs, &
684 plan%A_rt, plan%count_recv, plan%displ_recv, plan%ddtype_Bs, &
685 plan%ptdma_world, request(1), ierr)
686 call mpi_ialltoallw(plan%B_rd, plan%count_send, plan%displ_send, plan%ddtype_Fs, &
687 plan%B_rt, plan%count_recv, plan%displ_recv, plan%ddtype_Bs, &
688 plan%ptdma_world, request(2), ierr)
689 call mpi_ialltoallw(plan%C_rd, plan%count_send, plan%displ_send, plan%ddtype_Fs, &
690 plan%C_rt, plan%count_recv, plan%displ_recv, plan%ddtype_Bs, &
691 plan%ptdma_world, request(3), ierr)
692 call mpi_ialltoallw(plan%D_rd, plan%count_send, plan%displ_send, plan%ddtype_Fs, &
693 plan%D_rt, plan%count_recv, plan%displ_recv, plan%ddtype_Bs, &
694 plan%ptdma_world, request(4), ierr)
695
696 call mpi_waitall(4, request, mpi_statuses_ignore, ierr)
697
698 ! Solve the reduced cyclic tridiagonal systems of equations using cyclic TDMA.
699 call tdma_cycl_many(plan%A_rt, plan%B_rt, plan%C_rt, plan%D_rt, plan%n_sys_rt, plan%n_row_rt)
700
701 ! Transpose the obtained solutions to original reduced forms using MPI_Ialltoallw and DDTs.
702 call mpi_ialltoallw(plan%D_rt, plan%count_recv, plan%displ_recv, plan%ddtype_Bs, &
703 plan%D_rd, plan%count_send, plan%displ_send, plan%ddtype_Fs, &
704 plan%ptdma_world, request(1), ierr)
705 call mpi_waitall(1, request, mpi_statuses_ignore, ierr)
706
707 ! Update solutions of the modified tridiagonal system with the solutions of the reduced tridiagonal system.
708 do i=1,n_sys
709 d(i,1 ) = plan%D_rd(i,1)
710 d(i,n_row) = plan%D_rd(i,2)
711 enddo
712
713 do j=2,n_row-1
714 do i=1,n_sys
715 d(i,j) = d(i,j)-a(i,j)*d(i,1)-c(i,j)*d(i,n_row)
716 enddo
717 enddo
718 endif
719
720 end subroutine pascal_tdma_many_solve_cycle

722end module pascal_tdma
