230 subroutine pascal_tdma_many_solve_cuda(p, a_d, b_d, c_d, d_d)

237
238 if(p%nprocs.eq.1) then
239 ! Solve the tridiagonal system directly when nprocs = 1.
240 ! The size of shared memory is specified using shared_buffer_size

332 attributes(global) subroutine pascal_tdma_many_modified_thomas_cuda(a, b, c, d, a_rd, b_rd, c_rd, d_rd, nz_row)
333
334 implicit none
335
336 integer, value, intent(in) :: nz_row
337 double precision, device, intent(inout) :: a(:, :, :), c(:, :, :), d(:, :, :)
338 double precision, device, intent(in) :: b(:, :, :)
339 double precision, device, intent(inout) :: a_rd(:, :, :), b_rd(:, :, :)
340 double precision, device, intent(inout) :: c_rd(:, :, :), d_rd(:, :, :)
341
342 ! Temporary variables for computation
343 integer :: i, j, k
344 integer :: ti, tj, tk
345 double precision :: r
346
347 ! Block shared memory for pipeline copy
348 double precision, shared :: a1(blockdim%x+1, blockdim%y), a0(blockdim%x+1, blockdim%y)
349 double precision, shared :: b1(blockdim%x+1, blockdim%y), b0(blockdim%x+1, blockdim%y)
350 double precision, shared :: c1(blockdim%x+1, blockdim%y), c0(blockdim%x+1, blockdim%y)
351 double precision, shared :: d1(blockdim%x+1, blockdim%y), d0(blockdim%x+1, blockdim%y)
352 double precision, shared :: r0(blockdim%x+1, blockdim%y)
353
354 ! Global index
355 j = (blockidx%x-1) * blockdim%x + threadidx%x
356 k = (blockidx%y-1) * blockdim%y + threadidx%y
357
358 ! Local index in block
359 tj = threadidx%x
360 tk = threadidx%y
361
362 ! The modified Thomas algorithm : elimination of lower diagonal elements.
363 ! First & second indices indicate a number of independent many tridiagonal systems for parallezation.
364 ! Third index indicates a row number in a partitioned tridiagonal system .
365 ! Therefore, first & second indices are for thread IDs.
366 ! The modified Thomas algorithm : elimination of lower diagonal elements.
367
368 a0(tj, tk) = a(j, k, 1)
369 b0(tj, tk) = b(j, k, 1)
370 c0(tj, tk) = c(j, k, 1)
371 d0(tj, tk) = d(j, k, 1)
372
373 a0(tj, tk) = a0(tj, tk) / b0(tj, tk)
374 c0(tj, tk) = c0(tj, tk) / b0(tj, tk)
375 d0(tj, tk) = d0(tj, tk) / b0(tj, tk)
376
377 a(j, k,1) = a0(tj, tk)
378 c(j, k,1) = c0(tj, tk)
379 d(j, k,1) = d0(tj, tk)
380
381 a1(tj, tk) = a(j, k, 2)
382 b1(tj, tk) = b(j, k, 2)
383 c1(tj, tk) = c(j, k, 2)
384 d1(tj, tk) = d(j, k, 2)
385
386 a1(tj, tk) = a1(tj, tk) / b1(tj, tk)
387 c1(tj, tk) = c1(tj, tk) / b1(tj, tk)
388 d1(tj, tk) = d1(tj, tk) / b1(tj, tk)
389
390 a(j, k,2) = a1(tj, tk)
391 c(j, k,2) = c1(tj, tk)
392 d(j, k,2) = d1(tj, tk)
393
394 do i = 3, nz_row
395
396 ! Pipeline copy of (i-1)th data using shared memory
397 a0(tj, tk) = a1(tj, tk)
398 c0(tj, tk) = c1(tj, tk)
399 d0(tj, tk) = d1(tj, tk)
400
401 ! Load i-th data from global memory
402 a1(tj, tk) = a(j, k,i)
403 b1(tj, tk) = b(j, k,i)
404 c1(tj, tk) = c(j, k,i)
405 d1(tj, tk) = d(j, k,i)
406
407 r0(tj, tk) = 1.0d0 / (b1(tj, tk) - a1(tj, tk) * c0(tj, tk))
408 d1(tj, tk) = r0(tj, tk) * (d1(tj, tk) - a1(tj, tk) * d0(tj, tk))
409 c1(tj, tk) = r0(tj, tk) * c1(tj, tk)
410 a1(tj, tk) = -r0(tj, tk) * a1(tj, tk) * a0(tj, tk)
411
412 ! Save updated i-th data to global memory
413 a(j, k,i) = a1(tj, tk)
414 c(j, k,i) = c1(tj, tk)
415 d(j, k,i) = d1(tj, tk)
416
417 enddo
418
419 ! Construct many reduced tridiagonal systems per each rank. Each process has two rows of reduced systems.
420 a_rd(j, k,2) = a1(tj, tk)
421 b_rd(j, k,2) = 1.0d0
422 c_rd(j, k,2) = c1(tj, tk)
423 d_rd(j, k,2) = d1(tj, tk)
424
425 ! Pipeline copy between arrays in shared memory
426 a1(tj, tk) = a0(tj, tk)
427 c1(tj, tk) = c0(tj, tk)
428 d1(tj, tk) = d0(tj, tk)
429
430 ! The modified Thomas algorithm : elimination of upper diagonal elements.
431 do i = nz_row-2, 2, -1
432
433 ! Load i-th data from global memory
434 a0(tj, tk) = a(j, k,i)
435 c0(tj, tk) = c(j, k,i)
436 d0(tj, tk) = d(j, k,i)
437
438 d0(tj, tk) = d0(tj, tk) - c0(tj, tk)*d1(tj, tk)
439 a0(tj, tk) = a0(tj, tk) - c0(tj, tk)*a1(tj, tk)
440 c0(tj, tk) =-c0(tj, tk) * c1(tj, tk)
441
442 ! Pipeline copy of updated i-th data using shared memory
443 a1(tj, tk) = a0(tj, tk)
444 c1(tj, tk) = c0(tj, tk)
445 d1(tj, tk) = d0(tj, tk)
446
447 ! Save updated data to global memory
448 a(j, k,i) = a0(tj, tk)
449 c(j, k,i) = c0(tj, tk)
450 d(j, k,i) = d0(tj, tk)
451 enddo
452
453 a0(tj, tk) = a(j, k,1)
454 c0(tj, tk) = c(j, k,1)
455 d0(tj, tk) = d(j, k,1)
456
457 r0(tj, tk) = 1.0d0 / (1.0d0 - a1(tj, tk) * c0(tj, tk))
458 d0(tj, tk) = r0(tj, tk) * (d0(tj, tk) - c0(tj, tk) * d1(tj, tk))
459 a0(tj, tk) = r0(tj, tk) * a0(tj, tk)
460 c0(tj, tk) = -r0(tj, tk) * c0(tj, tk) * c1(tj, tk)
461
462 d(j, k,1) = d0(tj, tk)
463 a(j, k,1) = a0(tj, tk)
464 c(j, k,1) = c0(tj, tk)
465
466 ! Construct many reduced tridiagonal systems per each rank. Each process has two rows of reduced systems.
467 a_rd(j, k,1) = a0(tj, tk)
468 b_rd(j, k,1) = 1.0d0
469 c_rd(j, k,1) = c0(tj, tk)
470 d_rd(j, k,1) = d0(tj, tk)
471
472
473 end subroutine pascal_tdma_many_modified_thomas_cuda

483 attributes(global) subroutine pascal_tdma_many_update_solution_cuda(a, c, d, d_rd, nz_row)
484
485 implicit none
486
487 integer, value, intent(in) :: nz_row
488 double precision, device, intent(in) :: a(:, :, :), c(:, :, :), d_rd(:, :, :)
489 double precision, device, intent(inout) :: d(:, :, :)
490
491 ! Temporary variables for computation
492 integer :: i, j, k
493 integer :: tj, tk
494
495 ! Block shared memory
496 double precision, shared :: ds(blockdim%x + 1, blockdim%y), de(blockdim%x + 1, blockdim%y)
497
498 ! Global index
499 j = (blockidx%x - 1) * blockdim%x + threadidx%x
500 k = (blockidx%y - 1) * blockdim%y + threadidx%y
501
502 ! Local index in block
503 tj = threadidx%x
504 tk = threadidx%y
505
506 ! Using shared memory
507 ! First and second indices are for thread IDs
508 ds(tj, tk) = d_rd(j, k, 1)
509 de(tj, tk) = d_rd(j, k, 2)
510 call syncthreads()
511
512 ! Update solutions of the modified tridiagonal system with the solutions of the reduced tridiagonal system.
513 d(j, k, 1) = ds(tj, tk)
514 d(j, k, nz_row) = de(tj, tk)
515
516 do i = 2, nz_row-1
517 d(j, k, i) = d(j, k, i) - a(j, k, i) * ds(tj, tk) - c(j, k, i) * de(tj, tk)
518 enddo
519
520 end subroutine pascal_tdma_many_update_solution_cuda

562 attributes(global) subroutine mem_detach_slab_xy(slab_xy, array1D, n1, n2, n3, nprocs)
563
564 implicit none
565
566 integer, value, intent(in) :: n1, n2, n3, nprocs
567 double precision, device, intent(in) :: slab_xy(:, :, :)
568 double precision, device, intent(out) :: array1d(:)
569
570 ! Variables to calculate indices for in and out arrays
571 integer :: i, j, k, kblk, n1blksize, blksize
572 integer :: pos, pos_i, pos_j, pos_k
573
574 n1blksize = n1 / nprocs
575 blksize = n1blksize * n2 * n3
576
577 i = (blockidx%x - 1) * blockdim%x + threadidx%x
578 j = (blockidx%y - 1) * blockdim%y + threadidx%y
